	\begin{algorithm}
		\caption{Genetic algorithm guided by Bayesian Optimization}
		\label{array-sum}
		\begin{algorithmic}[1]
			\State {Initialize P(0) randomly}
			\For {$t=0$ to MaxGenerations}
			\State {Evaluate population P(t)}
			\State {Update the probabilistic model}
			\State {Selection()}
			\State {Crossover()}
			\State {Mutate()}
			\State {Form $P(t+1)$}
			\State {Get expected improvement for each individual in $P(t+1)$}
			\State {Remove $N*r_i$ individuals based on expected improvement}
			\State {Add $N*r_i$ elites from $P(t)$ to $P(t+1)$ where $r_i$ is the elitism ratio}
			\EndFor
		\end{algorithmic}
	\end{algorithm}
